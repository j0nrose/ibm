# IBM Qiskit - Bell State Creation and Measurement
# Demonstrates quantum entanglement and superposition

from qiskit import QuantumCircuit, ClassicalRegister, QuantumRegister
from qiskit import execute, Aer
from qiskit.visualization import plot_histogram
import matplotlib.pyplot as plt
import numpy as np

# Create quantum and classical registers
def create_bell_state_circuit():
    # Initialize 2 qubits and 2 classical bits
    qreg = QuantumRegister(2, 'q')
    creg = ClassicalRegister(2, 'c')
    circuit = QuantumCircuit(qreg, creg)
    
    # Create Bell state |Œ¶+‚ü© = (|00‚ü© + |11‚ü©)/‚àö2
    circuit.h(qreg[0])        # Hadamard gate on qubit 0
    circuit.cx(qreg[0], qreg[1])  # CNOT gate (control: q0, target: q1)
    
    # Add measurement
    circuit.measure(qreg, creg)
    
    return circuit

# Execute the quantum circuit
def run_quantum_experiment():
    # Create the circuit
    bell_circuit = create_bell_state_circuit()
    
    # Print circuit diagram
    print("Quantum Circuit:")
    print(bell_circuit)
    
    # Use Aer's QASM simulator
    simulator = Aer.get_backend('qasm_simulator')
    
    # Execute the circuit 1000 times
    job = execute(bell_circuit, simulator, shots=1000)
    result = job.result()
    
    # Get measurement counts
    counts = result.get_counts(bell_circuit)
    
    print(f"\nMeasurement Results (1000 shots):")
    for state, count in counts.items():
        probability = count / 1000
        print(f"State |{state}‚ü©: {count} times ({probability:.1%})")
    
    return counts

# Run the experiment
if __name__ == "__main__":
    print("=== IBM Qiskit Bell State Experiment ===")
    print("Creating entangled quantum state...")
    
    results = run_quantum_experiment()
    
    print("\n=== Analysis ===")
    print("Expected: ~50% |00‚ü© and ~50% |11‚ü©")
    print("This demonstrates quantum entanglement!")

üåü About This Quantum Circuit

This IBM Qiskit script creates a Bell State - one of the most famous quantum entangled states. Here's what happens:

Circuit Steps:
1. H gate ‚Üí Creates superposition: |0‚ü© + |1‚ü©
2. CNOT ‚Üí Entangles qubits: |00‚ü© + |11‚ü©
3. Measure ‚Üí Collapses to either |00‚ü© or |11‚ü©

‚öõÔ∏è Quantum Concepts Demonstrated

Superposition: The Hadamard gate puts qubit 0 into a superposition of |0‚ü© and |1‚ü© states simultaneously.

Entanglement: The CNOT gate creates quantum entanglement between the two qubits. When measured, they always show correlated results.

Measurement: Quantum states collapse upon measurement, giving either |00‚ü© or |11‚ü© with equal probability (~50% each).

üî¨ Expected Results

In an ideal quantum computer, this circuit produces:

  • ~50% chance of measuring |00‚ü©
  • ~50% chance of measuring |11‚ü©
  • ~0% chance of |01‚ü© or |10‚ü© (due to entanglement)

This demonstrates that the qubits are quantum mechanically entangled - measuring one instantly determines the state of the other, regardless of distance.
